import typing as t

def binary_search(seq: t.Sequece[int], target: int) -> int:
    left = 0
    right = len(seq) - 1
    while left < right:
        mid = (left + right) // 2
        if seq[mid] < target:
            left = mid + 1
        else:
            right = mid

        return mid
